    function escapeHtml(value) {
      return String(value ?? "")
        .replaceAll("&", "&amp;")
        .replaceAll("<", "&lt;")
        .replaceAll(">", "&gt;")
        .replaceAll('"', "&quot;")
        .replaceAll("'", "&#039;");
    }

    function sanitizeGuideHtml(html) {
      const allowedTags = new Set(["a", "blockquote", "br", "code", "em", "h1", "h2", "h3", "h4", "hr", "li", "ol", "p", "pre", "strong", "table", "tbody", "td", "th", "thead", "tr", "ul"]);
      const allowedAttrs = new Map([["a", new Set(["href", "title"])], ["th", new Set(["align"])], ["td", new Set(["align"])]]);
      const doc = new DOMParser().parseFromString(`<div>${html}</div>`, "text/html");
      const root = doc.body.firstElementChild;
      const clean = (node) => {
        Array.from(node.children).forEach((child) => {
          const tag = child.tagName.toLowerCase();
          if (!allowedTags.has(tag)) {
            child.replaceWith(doc.createTextNode(child.textContent || ""));
            return;
          }
          Array.from(child.attributes).forEach((attr) => {
            const name = attr.name.toLowerCase();
            const allowed = allowedAttrs.get(tag);
            if (!allowed || !allowed.has(name)) {
              child.removeAttribute(attr.name);
              return;
            }
            if (tag === "a" && name === "href") {
              const href = attr.value.trim();
              if (!/^(https?:|mailto:|#)/i.test(href)) child.removeAttribute(attr.name);
            }
          });
          if (tag === "a" && child.hasAttribute("href")) {
            child.setAttribute("target", "_blank");
            child.setAttribute("rel", "noopener noreferrer");
          }
          clean(child);
        });
      };
      clean(root);
      return root.innerHTML;
    }

    function renderGuideMarkdown(value) {
      const source = String(value ?? "");
      if (!window.marked || typeof window.marked.parse !== "function") {
        return `<pre>${escapeHtml(source)}</pre>`;
      }
      return sanitizeGuideHtml(window.marked.parse(source, {gfm: true, breaks: true}));
    }

    async function loadLeaderboard(gameId) {

        game.ranks = [];
        if (selected === game.id) renderRanks(game);

      try {
        const soupToken = localStorage.getItem(SOUP_TOKEN_KEY) || "";
        const res = await fetch("/soup/api/leaderboard/games", {
          headers: soupToken ? {Authorization: `Bearer ${soupToken}`} : {}
        });
        if (!res.ok) throw new Error("leaderboard unavailable");
        const rows = await res.json();
        game.ranks = Array.isArray(rows)
          ? rows.filter((row) => Number(row.score) > 0).slice(0, 3)
          : [];
      } catch (err) {
        game.ranks = [];
      }
      if (selected === game.id) renderRanks(game);
    }

      if (game === "love") return ({A: "肯定言辞", B: "优质时间", C: "服务行动", D: "赠送礼物", E: "身体接触"})[result] || result;
      if (game === "ecr") return ({secure: "安全型", fearful: "恐惧型", preoccupied: "迷恋型", dismissive: "冷漠型"})[result] || result;
      if (game === "humanity") return ({certified_carbon: "认证碳基", human_flavor: "人味充足", mixed_signal: "混合信号", cyber_infiltration: "赛博渗透中", check_cooling: "建议自查散热"})[result] || result;
      return result;
    }

    function renderDistribution(game, rows) {
      const safeRows = Array.isArray(rows) ? rows : [];
      const sortedRows = safeRows.slice().sort(
        (left, right) => (Number(right.count) || 0) - (Number(left.count) || 0)
      );
      const max = Math.max(1, ...sortedRows.map((row) => Number(row.count) || 0));
      const body = sortedRows.length
        ? sortedRows.map((row) => {
            const count = Number(row.count) || 0;
            const width = Math.max(4, (count / max) * 100);
            return `
              <div class="platform-dist-row">
                <span class="platform-dist-label">${escapeHtml(platformResultLabel(game, row.result))}</span>
                <span class="platform-dist-track" aria-hidden="true"><span style="width:${width}%"></span></span>
                <b>${count}</b>
              </div>
            `;
          }).join("")
        : `<p class="history-empty">暂无数据</p>`;
      return `
        <section class="platform-dist-block platform-dist-${game}">
          <div class="platform-dist-title">${escapeHtml(platformTestLabels[game] || game)}</div>
          ${body}
        </section>
      `;
    }

    function renderPlatformStats() {
      const content = $("leaderboardContent");
      if (platformStatsState.loading) {
        $("leaderboardMsg").textContent = "";
        content.innerHTML = `<p class="history-empty">统计加载中...</p>`;
        return;
      }
      if (platformStatsState.error) {
        $("leaderboardMsg").textContent = platformStatsState.error;
        content.innerHTML = `<p class="history-empty">暂时无法读取平台统计。</p>`;
        return;
      }
      const stats = platformStatsState.data;
      if (!stats) {
        $("leaderboardMsg").textContent = "";
        content.innerHTML = `<p class="history-empty">暂无数据</p>`;
        return;
      }
      const saveCounts = stats.save_counts || {};
      const distributions = stats.test_result_distributions || {};
      const cards = Object.entries(platformSummaryLabels).map(([key, label]) => `
        <div class="platform-stat-card">
          <span>${escapeHtml(label)}</span>
          <b>${Number(saveCounts[key] || 0)}</b>
        </div>
      `).join("");
      const dist = Object.keys(platformTestLabels).map((game) => renderDistribution(game, distributions[game])).join("");
      $("leaderboardMsg").textContent = "";
      content.innerHTML = `
        <div class="platform-stats-grid">${cards}</div>
        <div class="platform-stat-note">仅展示聚合统计，不包含玩家 ID、用户 ID 或用户名。</div>
        <div class="platform-distributions">${dist}</div>
        <div class="platform-entertainment-badge">仅供娱乐</div>
      `;
    }

    async function openPlatformStats() {
      closeDrawers();
      openModal("leaderboardModal");
      if (platformStatsState.data) {
        renderPlatformStats();
        return;
      }
      platformStatsState = {data: null, loading: true, error: ""};
      renderPlatformStats();
      try {
        const res = await fetch("/soup/api/leaderboard/platform-stats", {cache: "no-store"});

    }

    function ensureArcadeTarget() {
      const bindings = aiBindings();
      if (!bindings.length) {
        arcadeBankState.aiUserId = "";
        arcadeBankState.status = null;
        return null;
      }

    async function loadArcadeBankStatus() {
      if (currentGame().id !== "arcade" || !token()) {
        renderArcadeBanks();
        return;
      }
      const aiUserId = ensureArcadeTarget();
      if (!aiUserId) {
        renderArcadeBanks();
        return;
      }
      arcadeBankState.loading = true;
      renderArcadeBanks();
      try {
        const res = await fetch(`/api/arcade/chips?ai_user_id=${encodeURIComponent(aiUserId)}`, {
          headers: headers(false),
          cache: "no-store"
        });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(data.error || "读取失败");
        arcadeBankState.status = data;
        arcadeBankState.message = "";
      } catch (err) {
        arcadeBankState.status = null;
        arcadeBankState.message = err.message;
      } finally {
        arcadeBankState.loading = false;
        renderArcadeBanks();
      }
    }

    async function grantArcadeChips() {
      const aiUserId = ensureArcadeTarget();
      const amount = Number(arcadeBankState.amount);
      if (!aiUserId) {
        arcadeBankState.message = "请先绑定小机。";
        renderArcadeBanks();
        return;
      }
      if (!Number.isInteger(amount) || amount < 1) {
        arcadeBankState.message = "金额须为正整数。";
        renderArcadeBanks();
        return;
      }
      arcadeBankState.saving = true;
      arcadeBankState.message = "发放中...";
      renderArcadeBanks();
      try {
        const res = await fetch("/api/arcade/chips", {
          method: "POST",
          headers: headers(true),
          body: JSON.stringify({ai_user_id: Number(aiUserId), amount})
        });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(data.error || "发放失败");
        arcadeBankState.status = data;
        arcadeBankState.message = `已发放 ${amount} 筹码。`;
        showToast(`已发放 ${amount} 筹码`);
        await loadGameStats();
      } catch (err) {
        arcadeBankState.message = err.message;
      } finally {
        arcadeBankState.saving = false;
        renderArcadeBanks();
      }
    }

        machineState = {loaded: true, loading: false, machines: [], error: ""};

      } catch (err) {
        machineState = {loaded: true, loading: false, machines: [], error: err.message || "读取小机设置失败"};
      }
    }

    function bindMachinePanel() {
      document.querySelectorAll(".machine-card").forEach((card) => {
        const checkbox = card.querySelector(".machine-enabled");
        if (checkbox) {
          checkbox.addEventListener("change", () => {
            card.classList.toggle("enabled", checkbox.checked);
            const msg = card.querySelector(".machine-msg");
            if (checkbox.checked && msg) msg.textContent = "";
            if (!checkbox.checked) {
              checkbox.disabled = true;
              saveMachineSettings(card).then((saved) => {
                checkbox.disabled = false;
                if (!saved) {
                  checkbox.checked = true;
                  card.classList.add("enabled");
                }
              });
            }
          });
        }
        const save = card.querySelector(".machine-save");
        if (save) {
          save.addEventListener("click", () => saveMachineSettings(card));
        }
        const reset = card.querySelector(".machine-reset");
        if (reset) {
          reset.addEventListener("click", () => resetMachineSettings(card));
        }
      });
    }

    async function saveMachineSettings(card) {
      const aiUserId = Number(card.dataset.machineId);
      const msg = card.querySelector(".machine-msg");
      const payload = {
        ai_user_id: aiUserId,
        enabled: Boolean(card.querySelector(".machine-enabled")?.checked),
        remind_threshold: Number(card.querySelector(".machine-remind")?.value || 30),
        force_threshold: Number(card.querySelector(".machine-force")?.value || 50),
        lock_minutes: Number(card.querySelector(".machine-lock-minutes")?.value || 30),
        allow_self_reset: Boolean(card.querySelector(".machine-self-reset")?.checked),
      };
      if (msg) msg.textContent = "保存中...";
      try {
        const res = await fetch("/api/anti-addiction/settings", {
          method: "POST",
          headers: headers(true),
          body: JSON.stringify(payload),
        });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(data.error || "保存失败");
        const idx = machineState.machines.findIndex((item) => Number(item.id) === aiUserId);
        if (idx >= 0) machineState.machines[idx].anti_addiction = data.anti_addiction;
        if (msg) msg.textContent = "已保存";
        return true;
      } catch (err) {
        if (msg) msg.textContent = err.message || "保存失败";
        return false;
      }
    }

    async function resetMachineSettings(card) {
      const aiUserId = Number(card.dataset.machineId);
      const msg = card.querySelector(".machine-msg");
      if (msg) msg.textContent = "重置中...";
      try {
        const res = await fetch("/api/anti-addiction/reset", {
          method: "POST",
          headers: headers(true),
          body: JSON.stringify({ai_user_id: aiUserId}),
        });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(data.error || "重置失败");
        if (msg) msg.textContent = "已重置";
      } catch (err) {
        if (msg) msg.textContent = err.message || "重置失败";
      }
    }

    async function loadMe() {
      loadMeAbort?.abort();
      const controller = new AbortController();
      loadMeAbort = controller;
      if (!token()) {
        me = null;
        renderAuth();
        return;
      }
      try {
        const res = await fetch("/api/auth/me", {
          headers: headers(false),
          signal: controller.signal,
        });
        if (!res.ok) throw new Error("auth failed");
        if (controller.signal.aborted) return;
        me = await res.json();
        machineState = {loaded: false, loading: false, machines: [], error: ""};
        if (me && me.user && me.user.id) {
          localStorage.setItem("cedartoy_user_id", me.user.id);
        }
      } catch (err) {
        if (err.name === "AbortError") return;
        localStorage.removeItem(TOKEN_KEY);
        localStorage.removeItem("cedartoy_user_id");
        me = null;
      }
      if (controller.signal.aborted) return;
      renderAuth();
      renderGames();
      renderDetail();
      if (currentGame().id === "arcade") loadArcadeBankStatus();
    }

    const USERNAME_RE = /^[a-zA-Z0-9_\u4e00-\u9fff]{2,20}$/;

    function validateLoginInput(username, password) {
      if (!username || !password) return "用户名和密码必填";
      if (username.length < 2 || username.length > 20) return "用户名长度须为 2-20 个字符";
      if (!USERNAME_RE.test(username)) return "用户名只能包含字母、数字、下划线和中文";
      if (password.length < 6) return "密码至少 6 位";
      return "";
    }

    async function login() {
      loadMeAbort?.abort();
      $("loginMsg").textContent = "";
      const username = $("loginUser").value.trim();
      const password = $("loginPass").value;
      const validationError = validateLoginInput(username, password);
      if (validationError) {
        $("loginMsg").textContent = validationError;
        return;
      }
      try {
        const res = await fetch("/api/auth/login_or_register", {
          method: "POST",
          headers: {"Content-Type": "application/json"},
          body: JSON.stringify({username, password})
        });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(data.error || "登录失败");
        localStorage.setItem(TOKEN_KEY, data.token);
        localStorage.setItem("cedartoy_user_id", data.user.id);
        me = {user: data.user, bindings: []};
        machineState = {loaded: false, loading: false, machines: [], error: ""};
        closeModals();
        renderAuth();
        renderGames();
        renderDetail();
        if (currentGame().id === "arcade") loadArcadeBankStatus();
      } catch (err) {
        $("loginMsg").textContent = err.message;
      }
    }

    async function bindAccount() {
      $("bindMsg").textContent = "";
      const binding_token = $("bindToken").value.trim();
      if (!binding_token) {
        $("bindMsg").textContent = "绑定码必填";
        return;
      }
      try {
        const res = await fetch("/api/auth/bind", {
          method: "POST",
          headers: headers(true),
          body: JSON.stringify({binding_token})
        });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(data.error || "绑定失败");
        $("bindMsg").textContent = "绑定成功";
        await loadMe();
        machineState = {loaded: false, loading: false, machines: [], error: ""};
        renderMine();
        setTimeout(closeModals, 500);
      } catch (err) {
        $("bindMsg").textContent = err.message;
      }
    }

    async function loadRoomCount() {
      try {
        const soupToken = localStorage.getItem(SOUP_TOKEN_KEY) || "";

          headers: soupToken ? {Authorization: `Bearer ${soupToken}`} : {}

        const rooms = await res.json();

          ? rooms.filter((room) => room.status === "waiting" || room.status === "playing").length

    function syncViewportInsets() {
      const vv = window.visualViewport;
      if (!vv) return;
      const inset = Math.max(0, Math.round(window.innerHeight - vv.height - vv.offsetTop));
      document.documentElement.style.setProperty("--browser-bottom", `${inset}px`);
    }

    if (window.visualViewport) {
      window.visualViewport.addEventListener("resize", syncViewportInsets);
      window.visualViewport.addEventListener("scroll", syncViewportInsets);
      window.addEventListener("resize", syncViewportInsets);
      syncViewportInsets();
    }
